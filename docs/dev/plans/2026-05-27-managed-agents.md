# Managed Agents (v1) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a named, versioned, workspace-private `AgentDefinition` that resolves into a run by overriding the workspace-singleton prompt/model/MCP/skill/sandbox lookups in `_run_cubepi_path`, with the conversation pinning an immutable version on its first turn.

**Architecture:** A mutable `AgentDefinition` head + immutable `AgentDefinitionVersion` rows (mirrors `Skill`/`SkillVersion`). Two association tables pin `mcp_connector_install_id` and a concrete `skill_version_id` (+ the `org_skill_install_id` it was selected through). A dedicated `AgentDefinitionRepository` (constructor takes `org_id`, like `MCPConnectorInstallRepository` — NOT `ScopedRepository`) handles the nullable `workspace_id`. A new `AgentResolver` service maps a version to a frozen `ResolvedAgentConfig` dataclass that threads through `_run_cubepi_path`, replacing the singleton reads. `start_run` gains `agent_definition_id` + optional `version_id`; the conversation pins the resolved version on the first turn. Routes are workspace-scoped only (`/api/v1/ws/{ws}/agents/...`). Resolution verifies each pinned `skill_version_id` belongs to a visible `OrgSkillInstall` in the caller's org before calling `fetch_skill_md` (which has no scope check).

**Tech Stack:** FastAPI, SQLModel + Alembic (autogenerate), cubepi runtime, Postgres, pytest E2E (`tests/e2e/`), mypy strict, 100-char lines.

```python
"""Managed Agent models — definition head, immutable version, and ref tables."""

from datetime import datetime
from typing import Any, ClassVar

import sqlalchemy as sa
from sqlalchemy import JSON, Column, Index, UniqueConstraint
from sqlmodel import Field, SQLModel

from cubebox.models.mixins import CubeboxBase, TimestampMixin
from cubebox.models.public_id import (
    PREFIX_AGENT_DEFINITION,
    PREFIX_AGENT_DEFINITION_VERSION,
)


class AgentDefinition(CubeboxBase, table=True):
    """Mutable head. workspace_id NULL ⇒ org-shared (deferred to v2); v1 always sets it."""

    _PREFIX: ClassVar[str] = PREFIX_AGENT_DEFINITION
    __tablename__ = "agent_definitions"

    org_id: str = Field(foreign_key="organizations.id", max_length=20, index=True)
    workspace_id: str | None = Field(
        default=None, foreign_key="workspaces.id", max_length=20, nullable=True, index=True
    )
    name: str = Field(max_length=128)
    description: str = Field(max_length=1024, default="")
    current_version: str | None = Field(default=None, max_length=32)
    published_version: str | None = Field(default=None, max_length=32)
    created_by_user_id: str = Field(foreign_key="users.id", max_length=20)
    archived_at: datetime | None = Field(default=None)

    __table_args__ = (
        Index(
            "uq_agent_def_org_shared",
            "org_id",
            "name",
            unique=True,
            postgresql_where=sa.text("workspace_id IS NULL"),
        ),
        UniqueConstraint("org_id", "workspace_id", "name", name="uq_agent_def_ws"),
        Index("ix_agent_def_org_workspace", "org_id", "workspace_id"),
    )


class AgentDefinitionVersion(CubeboxBase, table=True):
    """Immutable, append-only. Carries the resolvable config snapshot."""

    _PREFIX: ClassVar[str] = PREFIX_AGENT_DEFINITION_VERSION
    __tablename__ = "agent_definition_versions"

    agent_definition_id: str = Field(
        foreign_key="agent_definitions.id", max_length=20, index=True
    )
    version: str = Field(max_length=32)
    created_by_user_id: str = Field(foreign_key="users.id", max_length=20)
    instructions: str = Field(sa_column=Column(sa.Text))
    model_ref: dict[str, Any] = Field(default_factory=dict, sa_column=Column(JSON))
    sandbox_policy: str = Field(max_length=32, default="default")  # "none" | "default"

    __table_args__ = (
        UniqueConstraint(
            "agent_definition_id", "version", name="uq_agent_def_version"
        ),
    )


class AgentDefinitionMcpRef(SQLModel, TimestampMixin, table=True):
    """Pins an MCP connector install onto a version. Composite PK; no public_id."""

    __tablename__ = "agent_definition_mcp_refs"

    agent_definition_version_id: str = Field(
        primary_key=True,
        foreign_key="agent_definition_versions.id",
        max_length=20,
        index=True,
    )
    mcp_connector_install_id: str = Field(
        primary_key=True,
        foreign_key="mcp_connector_installs.id",
        max_length=20,
        index=True,
    )


class AgentDefinitionSkillRef(SQLModel, TimestampMixin, table=True):
    """Pins a concrete SkillVersion onto a version (immutable snapshot).

    Stores the OrgSkillInstall it was selected through so the resolver can
    verify the pinned version is still installed/visible before binding —
    never trusting the stored skill_version_id alone.
    Composite PK; no public_id.
    """

    __tablename__ = "agent_definition_skill_refs"

    agent_definition_version_id: str = Field(
        primary_key=True,
        foreign_key="agent_definition_versions.id",
        max_length=20,
        index=True,
    )
    skill_version_id: str = Field(
        primary_key=True, foreign_key="skill_versions.id", max_length=20, index=True
    )
    org_skill_install_id: str = Field(
        foreign_key="org_skill_installs.id", max_length=20, index=True
    )
```

```python
"""Repository for managed agent definitions.

``workspace_id`` is nullable (NULL ⇒ org-shared, deferred to v2), so this repo
cannot derive from ScopedRepository — it follows the MCPConnectorInstallRepository
precedent: constructor takes ``org_id`` and enforces it on every query and add().
"""

from datetime import UTC, datetime

from sqlalchemy import or_, select
from sqlalchemy.ext.asyncio import AsyncSession

from cubebox.models import (
    AgentDefinition,
    AgentDefinitionMcpRef,
    AgentDefinitionSkillRef,
    AgentDefinitionVersion,
)


class AgentDefinitionRepository:
    def __init__(self, session: AsyncSession, *, org_id: str) -> None:
        self.session = session
        self.org_id = org_id

    async def get(self, definition_id: str) -> AgentDefinition | None:
        stmt = select(AgentDefinition).where(
            AgentDefinition.id == definition_id,  # type: ignore[arg-type]
            AgentDefinition.org_id == self.org_id,  # type: ignore[arg-type]
        )
        return (await self.session.execute(stmt)).scalar_one_or_none()

    async def list_workspace(self, workspace_id: str) -> list[AgentDefinition]:
        stmt = (
            select(AgentDefinition)
            .where(
                AgentDefinition.org_id == self.org_id,  # type: ignore[arg-type]
                AgentDefinition.workspace_id == workspace_id,  # type: ignore[arg-type]
                AgentDefinition.archived_at.is_(None),  # type: ignore[union-attr]
            )
            .order_by(AgentDefinition.name)  # type: ignore[arg-type]
        )
        return list((await self.session.execute(stmt)).scalars().all())

    async def list_org_shared(self) -> list[AgentDefinition]:
        stmt = (
            select(AgentDefinition)
            .where(
                AgentDefinition.org_id == self.org_id,  # type: ignore[arg-type]
                AgentDefinition.workspace_id.is_(None),  # type: ignore[union-attr]
                AgentDefinition.archived_at.is_(None),  # type: ignore[union-attr]
            )
            .order_by(AgentDefinition.name)  # type: ignore[arg-type]
        )
        return list((await self.session.execute(stmt)).scalars().all())

    async def list_visible(self, workspace_id: str) -> list[AgentDefinition]:
        """Workspace-authoring view: workspace-private rows ∪ org-shared rows."""
        stmt = (
            select(AgentDefinition)
            .where(
                AgentDefinition.org_id == self.org_id,  # type: ignore[arg-type]
                AgentDefinition.archived_at.is_(None),  # type: ignore[union-attr]
                or_(
                    AgentDefinition.workspace_id == workspace_id,
                    AgentDefinition.workspace_id.is_(None),  # type: ignore[union-attr]
                ),
            )
            .order_by(AgentDefinition.name)  # type: ignore[arg-type]
        )
        return list((await self.session.execute(stmt)).scalars().all())

    async def add(self, definition: AgentDefinition) -> AgentDefinition:
        definition.org_id = self.org_id  # defend against caller-supplied org
        self.session.add(definition)
        await self.session.commit()
        await self.session.refresh(definition)
        return definition

    async def update(self, definition: AgentDefinition) -> AgentDefinition:
        if definition.org_id != self.org_id:
            raise RuntimeError("AgentDefinitionRepository.update: different org")
        definition.updated_at = datetime.now(UTC)
        self.session.add(definition)
        await self.session.commit()
        await self.session.refresh(definition)
        return definition

    async def get_version(self, version_id: str) -> AgentDefinitionVersion | None:
        v = await self.session.get(AgentDefinitionVersion, version_id)
        if v is None:
            return None
        parent = await self.get(v.agent_definition_id)
        return v if parent is not None else None

    async def add_version(
        self,
        *,
        version: AgentDefinitionVersion,
        mcp_install_ids: list[str],
        skill_refs: list[tuple[str, str]],  # (skill_version_id, org_skill_install_id)
    ) -> AgentDefinitionVersion:
        self.session.add(version)
        await self.session.flush()
        for install_id in mcp_install_ids:
            self.session.add(
                AgentDefinitionMcpRef(
                    agent_definition_version_id=version.id,
                    mcp_connector_install_id=install_id,
                )
            )
        for skill_version_id, org_skill_install_id in skill_refs:
            self.session.add(
                AgentDefinitionSkillRef(
                    agent_definition_version_id=version.id,
                    skill_version_id=skill_version_id,
                    org_skill_install_id=org_skill_install_id,
                )
            )
        await self.session.commit()
        await self.session.refresh(version)
        return version

    async def list_mcp_refs(self, version_id: str) -> list[str]:
        stmt = select(AgentDefinitionMcpRef.mcp_connector_install_id).where(
            AgentDefinitionMcpRef.agent_definition_version_id == version_id  # type: ignore[arg-type]
        )
        return list((await self.session.execute(stmt)).scalars().all())

    async def list_skill_refs(self, version_id: str) -> list[AgentDefinitionSkillRef]:
        stmt = select(AgentDefinitionSkillRef).where(
            AgentDefinitionSkillRef.agent_definition_version_id == version_id  # type: ignore[arg-type]
        )
        return list((await self.session.execute(stmt)).scalars().all())
```

```python
"""Resolve an AgentDefinitionVersion into the inputs _run_cubepi_path reads.

A definition does not need a new runtime — it overrides the workspace-singleton
lookups (prompt / model / MCP filter / skill set / sandbox). The resolved config
is frozen for the run; the conversation pins the version so the cache prefix stays
byte-identical across turns.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from cubebox.config import config as _config
from cubebox.models import (
    OrgSkillInstall,
    SkillVersion,
)
from cubebox.prompts.skills import SKILLS_PROMPT_TEMPLATE
from cubebox.prompts.system import BASE_SYSTEM_PROMPT
from cubebox.repositories.agent_definition import AgentDefinitionRepository
from cubebox.skills.cache import SkillCache
from cubebox.skills.service import SkillCatalogService


class AgentResolveError(RuntimeError):
    """Definition or version not found / not visible to the caller."""


class SkillRefVisibilityError(AgentResolveError):
    """A pinned skill_version_id is not backed by a visible install in this org."""


@dataclass(frozen=True)
class ResolvedSkillRef:
    skill_version_id: str
    name: str
    description: str
    version: str


@dataclass(frozen=True)
class ResolvedAgentConfig:
    """Frozen, run-scoped agent assembly. Threaded through _run_cubepi_path."""

    definition_id: str
    pinned_version_id: str
    effective_system_prompt: str
    model_ref: dict[str, object]
    sandbox_policy: str  # "none" | "default"
    mcp_install_id_filter: set[str]
    skill_version_ids: list[str]
    resolved_skills: list[ResolvedSkillRef] = field(default_factory=list)


class AgentResolver:
    def __init__(self, *, session: AsyncSession, org_id: str, workspace_id: str) -> None:
        self.session = session
        self.org_id = org_id
        self.workspace_id = workspace_id
        self.repo = AgentDefinitionRepository(session, org_id=org_id)
        self.catalog = SkillCatalogService(
            session=session,
            cache=SkillCache(
                cache_root=Path(_config.get("skills.cache_root", "skills_cache"))
            ),
        )

    async def resolve(
        self, *, definition_id: str, version_id: str | None = None
    ) -> ResolvedAgentConfig:
        definition = await self.repo.get(definition_id)
        if definition is None:
            raise AgentResolveError(f"agent definition not visible: {definition_id}")

        # version_id pins directly; otherwise resolve the published version
        # *string* on the head to its concrete version row id.
        resolved_version_id = await self._resolve_version_id(definition_id, version_id)
        version = await self.repo.get_version(resolved_version_id)
        if version is None:
            raise AgentResolveError(f"agent version not visible: {resolved_version_id}")

        skill_refs = await self.repo.list_skill_refs(version.id)
        resolved_skills = await self._verify_and_load_skills(skill_refs)

        prompt = BASE_SYSTEM_PROMPT + "\n\n" + version.instructions
        if resolved_skills:
            skills_list = "\n".join(
                f"- `{s.name}` — {s.description}"
                for s in sorted(resolved_skills, key=lambda s: s.name)
            )
            prompt += "\n\n" + SKILLS_PROMPT_TEMPLATE.format(skills_list=skills_list)

        mcp_filter = set(await self.repo.list_mcp_refs(version.id))

        return ResolvedAgentConfig(
            definition_id=definition.id,
            pinned_version_id=version.id,
            effective_system_prompt=prompt,
            model_ref=dict(version.model_ref),
            sandbox_policy=version.sandbox_policy,
            mcp_install_id_filter=mcp_filter,
            skill_version_ids=[s.skill_version_id for s in resolved_skills],
            resolved_skills=resolved_skills,
        )

    async def _resolve_version_id(
        self, definition_id: str, version_id: str | None
    ) -> str:
        if version_id is not None:
            return version_id
        definition = await self.repo.get(definition_id)
        assert definition is not None
        if definition.published_version is None:
            raise AgentResolveError(
                f"agent definition has no published version: {definition_id}"
            )
        # published_version is a version string; find its row.
        from cubebox.models import AgentDefinitionVersion

        stmt = select(AgentDefinitionVersion.id).where(
            AgentDefinitionVersion.agent_definition_id == definition_id,  # type: ignore[arg-type]
            AgentDefinitionVersion.version == definition.published_version,  # type: ignore[arg-type]
        )
        row = (await self.session.execute(stmt)).scalar_one_or_none()
        if row is None:
            raise AgentResolveError(
                f"published version row missing for {definition_id}"
            )
        return row

    async def _verify_and_load_skills(self, skill_refs) -> list[ResolvedSkillRef]:  # type: ignore[no-untyped-def]
        resolved: list[ResolvedSkillRef] = []
        for ref in skill_refs:
            # 1. The pinned version must exist.
            sv = await self.session.get(SkillVersion, ref.skill_version_id)
            if sv is None:
                raise SkillRefVisibilityError(
                    f"pinned skill_version not found: {ref.skill_version_id}"
                )
            # 2. The install it was selected through must be visible in this org
            #    AND point at the same skill as the pinned version. fetch_skill_md
            #    has no scope check, so never trust the stored id alone.
            install = await self.session.get(OrgSkillInstall, ref.org_skill_install_id)
            if (
                install is None
                or install.org_id != self.org_id
                or install.skill_id != sv.skill_id
                or (
                    install.workspace_id is not None
                    and install.workspace_id != self.workspace_id
                )
            ):
                raise SkillRefVisibilityError(
                    f"pinned skill not visible in org/workspace: {ref.skill_version_id}"
                )
            resolved.append(
                ResolvedSkillRef(
                    skill_version_id=sv.id,
                    name=(await self._skill_name(sv.skill_id)),
                    description=sv.description,
                    version=sv.version,
                )
            )
        return resolved

    async def _skill_name(self, skill_id: str) -> str:
        from cubebox.models import Skill

        skill = await self.session.get(Skill, skill_id)
        return skill.name if skill is not None else skill_id
```

```python
"""Workspace-scoped managed-agent endpoints under /api/v1/ws/{workspace_id}/agents.

Workspace-private only (v1). Org sharing + /admin/agents are deferred to v2;
per CLAUDE.md scope isolation, the admin scope gets its own router later — never
a ?scope= param here.
"""

from __future__ import annotations

from typing import Annotated

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession

from cubebox.api.schemas.agent_definition import (
    AgentDefinitionSummary,
    AgentDefinitionVersionResponse,
    CreateAgentDefinitionRequest,
    PublishVersionRequest,
)
from cubebox.auth.context import RequestContext
from cubebox.auth.dependencies import require_member
from cubebox.db import get_session
from cubebox.models import AgentDefinition, AgentDefinitionVersion
from cubebox.repositories.agent_definition import AgentDefinitionRepository

router = APIRouter(prefix="/ws/{workspace_id}/agents", tags=["ws-agents"])


@router.get("", response_model=list[AgentDefinitionSummary])
async def list_agents(
    workspace_id: str,
    *,
    ctx: Annotated[RequestContext, Depends(require_member)],
    session: Annotated[AsyncSession, Depends(get_session)],
) -> list[AgentDefinitionSummary]:
    repo = AgentDefinitionRepository(session, org_id=ctx.org_id)
    rows = await repo.list_workspace(workspace_id)
    return [
        AgentDefinitionSummary(
            id=r.id,
            name=r.name,
            description=r.description,
            workspace_id=r.workspace_id,
            current_version=r.current_version,
            published_version=r.published_version,
        )
        for r in rows
    ]


@router.post("", response_model=AgentDefinitionSummary, status_code=status.HTTP_201_CREATED)
async def create_agent(
    workspace_id: str,
    body: CreateAgentDefinitionRequest,
    *,
    ctx: Annotated[RequestContext, Depends(require_member)],
    session: Annotated[AsyncSession, Depends(get_session)],
) -> AgentDefinitionSummary:
    repo = AgentDefinitionRepository(session, org_id=ctx.org_id)
    row = await repo.add(
        AgentDefinition(
            org_id=ctx.org_id,
            workspace_id=workspace_id,
            name=body.name,
            description=body.description,
            created_by_user_id=ctx.user.id,
        )
    )
    return AgentDefinitionSummary(
        id=row.id,
        name=row.name,
        description=row.description,
        workspace_id=row.workspace_id,
        current_version=row.current_version,
        published_version=row.published_version,
    )


@router.post(
    "/{agent_id}/versions",
    response_model=AgentDefinitionVersionResponse,
    status_code=status.HTTP_201_CREATED,
)
async def publish_version(
    workspace_id: str,
    agent_id: str,
    body: PublishVersionRequest,
    *,
    ctx: Annotated[RequestContext, Depends(require_member)],
    session: Annotated[AsyncSession, Depends(get_session)],
) -> AgentDefinitionVersionResponse:
    repo = AgentDefinitionRepository(session, org_id=ctx.org_id)
    definition = await repo.get(agent_id)
    if definition is None or definition.workspace_id != workspace_id:
        raise HTTPException(status_code=404, detail="agent definition not found")
    version = await repo.add_version(
        version=AgentDefinitionVersion(
            agent_definition_id=definition.id,
            version=body.version,
            created_by_user_id=ctx.user.id,
            instructions=body.instructions,
            model_ref=body.model_ref,
            sandbox_policy=body.sandbox_policy,
        ),
        mcp_install_ids=body.mcp_install_ids,
        skill_refs=[(s.skill_version_id, s.org_skill_install_id) for s in body.skill_refs],
    )
    definition.current_version = body.version
    if body.publish:
        definition.published_version = body.version
    await repo.update(definition)
    return AgentDefinitionVersionResponse(
        id=version.id,
        version=version.version,
        instructions=version.instructions,
        model_ref=version.model_ref,
        sandbox_policy=version.sandbox_policy,
    )
```
